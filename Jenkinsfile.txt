pipeline {
    agent any 
    stages {
        stage('clone repo') { 
            steps {
                sh "rm -rf My-drone"
                sh "git clone https://github.com/rehmanmohd524-pixel/pipeline.git"
                sh "mvn clean -f My-drone"
            }
        }
        stage('Compile') { 
            steps {
                sh "mvn compile -f My-drone"
            }
        }
        stage('Test') { 
            steps {
                sh "mvn test -f My-drone"
            }
        }
        stage('package') { 
            steps {
                sh "mvn package -f My-drone"
            }
        }
        stage('install') { 
            steps {
                sh "mvn install -f My-drone"
            }
        }
        stage('deploy') { 
            steps {
                sh "cp /root/.m2/repository/DevOps/TestProject/1.0/TestProject-1.0.war /var/lib/tomcat10/webapps"
            }
        }
    }
}
